git --help : trợ giúp, hướng dẫn
git --version : hiển thịthông tin phiên bản git
git status : hiển thị trạng thái kho lưu trữ
git log : hiển thị lịch sử commit
git init [repo name] : tạo ra 1 kho lưu trữ
git clone [repo name] [clone name] : tạo một bản sao lket với repo
git config -l : xem cấu hình hiện tại
git config -l [--scope] [option name] [value] :
   scope : --system -> tất cả ng dùng
           --global -> liên quan đến vài repo
           --local  -> liên quan đến 1 reop
